Use a Case statement and Message Box to test the value of something.


Sub TestingValues

Dim NumValues As Long


Select Case NumValues

Case Is > 10

MsgBox "Value is greater than 10"

Case 5 To 10

MsgBox "Value is between 5 and 10.

Case 1, 2, 3, 4

MsgBox "Value is between 1 and 4.

Case Else

MsgBox "Value is less than 1"

End Select


End Sub
